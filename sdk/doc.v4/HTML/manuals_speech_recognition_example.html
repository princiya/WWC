Example 116 shows a JavaScript program for speech recognition.

See Speech Recognition for programming concepts and procedures, and PXCMSession and PXCMSpeechRecognition for the list of supported functions, marked with the (JavaScript) suffix.

<head>

   <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/jquery/1.7.2/jquery.min.js"></script>

   <script type="text/javascript" src="realsense.js"></script>

 

   <script type="text/javascript">

       var speech;

 

       function SpeechStart() {

           PXCMSession_CreateInstance().then(function (session) {

               return session.CreateImpl(undefined, undefined, pxcConst.PXCMSpeechRecognition.CUID);

           }).then(function (instance) {

               speech=instance;

               return speech.SetDictation();

           }).then(function (result) {

               return speech.StartRec(OnData, OnAlert);

           }).then(function (result) {

           }).catch(function (error) {

           });

       }

 

       function SpeechStop() {

           speech.StopRec().then(function (result) {

           });

       );

 

       function OnData(data) {

       }

 

       function OnAlert(data) {

       }

   </script>

</head>
