PXCAudioSourceService

Description

The PXCAudioSourceService interface reads audio samples from an audio device for speech module development. You can derive the PXCAudioSourceService interface from the PXCAudioSource instance.

Reference

C++

service\pxcaudiosourceservice.h

Example

Example 156 shows how to read audio samples from an audio device.

C++ Example 156: Read Audio Samples

// Create the PXCSession, PXCAudioSource, and PXCAudioSourceService instances

PXCSession *session=PXCSession::CreateInstance();

PXCAudioSource *source=session->CreateAudioSource();

PXCAudioSourceService *ssource=source->QueryInstance<PXCAudioSourceService>();

 

// Setup the audio device

PXCAudioSource::DeviceInfo dinfo={};

source->QueryDeviceInfo(0, &dinfo);

source->SetDevice(&dinfo);

 

// Set the audio configuration.

PXCAudioSourceService::AudioProfile pinfo={};

pinfo.audioInfo.nchannels=2;

pinfo.audioInfo.format=PXCAudio::AUDIO_FORMAT_PCM;

pinfo.audioInfo.sampleRate=48000;

ssource->Open(&pinfo);

 

// Stream to read 200 audio samples

for (int i=0; i<200; i++) {

   // Read an audio sample

   PXCAudio *sample=0;

   pxcStatus sts=ssource->ReadSample(&sample);

   if (sts<PXC_STATUS_NO_ERROR) break;

   if (sample==NULL) continue;

 

   // Do something with the audio sample

   ...

 

   // Release the audio sample after use.

   sample->Release();

}

 

// Clean up

ssource->Close();

source->Release();

session->Release();
